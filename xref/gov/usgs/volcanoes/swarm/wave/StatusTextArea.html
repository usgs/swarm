2   
3   import java.util.TimeZone;
4   
5   import javax.swing.JTextArea;
6   
7   import gov.usgs.volcanoes.core.data.Wave;
8   import gov.usgs.volcanoes.core.time.J2kSec;
9   import gov.usgs.volcanoes.core.time.Time;
10  import gov.usgs.volcanoes.swarm.SwarmConfig;
11  
12  /**
13   * Holds and displays information about wave at selected time.
14   * 
15   * @author Diana Norgaard
16   *
17   */
18  public class StatusTextArea extends JTextArea {
19  
20    private static final long serialVersionUID = -4045063168343152079L;
21    private static SwarmConfig swarmConfig;
22    
23    /**
24     * Default Constructor.
25     */
26    public StatusTextArea(String text) {
27      super(text);
28      setEditable(false);
29      setLineWrap(true);
30      swarmConfig = SwarmConfig.getInstance();    
31    }  
32  
33    /**
34     * Perform duration calculation.
35     * 
36     * @param startTime duration start time in j2k
37     * @param endTime duration end time in j2k
38     * @return duration string
39     */
40    public static String getDuration(double startTime, double endTime) {
41      if (swarmConfig.durationEnabled) {
42        double duration = Math.abs(startTime - endTime);
43        double durationMagnitude = swarmConfig.getDurationMagnitude(duration);
44        return String.format("Duration: %.2fs (Md: %.2f)", duration, durationMagnitude);
45      } else {
46        return null;
47      }
48    }
49      
50    /**
51     * Convert time at cursor to String for display.
52     * @param time in j2k
53     * @return formatted string of time in local and GMT
54     */
55    public static String getTimeString(double time, TimeZone tz) {
56      String utc = J2kSec.format(Time.STANDARD_TIME_FORMAT_MS, time);
57      double tzo = tz.getOffset(J2kSec.asEpoch(time)) / 1000;
58      if (tzo != 0) {
59        String tza = tz.getDisplayName(tz.inDaylightTime(J2kSec.asDate(time)), TimeZone.SHORT);
60        String status = J2kSec.format(Time.STANDARD_TIME_FORMAT_MS, time + tzo) + " (" + tza + "), "
61            + utc + " (UTC)";
62        return status;
63      } else {
64        return utc;
65      }
66    }
67  
68    
69    /**
70     * Get wave information text for display.
71     * 
72     * @param wave wave to get information for
73     * @return wave information text
74     */
75    public static String getWaveInfo(Wave wave) {
76      int[] dataRange = wave.getDataRange();
77      String waveInfo = null;
78      try {
79        waveInfo = String.format("[%s - %s (UTC), %d samples (%.2f s), %d samples/s, %d, %d]",
80            J2kSec.format(Time.STANDARD_TIME_FORMAT_MS, wave.getStartTime()),
81            J2kSec.format(Time.STANDARD_TIME_FORMAT_MS, wave.getEndTime()), wave.numSamples(),
82            wave.numSamples() / wave.getSamplingRate(), (int) wave.getSamplingRate(), dataRange[0],
83            dataRange[1]);
84      } catch (NullPointerException e) {
85        // do nothing
86      }
87      return waveInfo;
88    }
89    
90  }
