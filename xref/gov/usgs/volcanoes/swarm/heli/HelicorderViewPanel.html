214   /**
215    * Mark time on inset wave panel.
216    * @param t time
217    */
218   public void markTime(double t) {
219     if (Double.isNaN(startMark) && Double.isNaN(endMark)) {
220       startMark = t;
221     } else if (!Double.isNaN(startMark) && Double.isNaN(endMark)) {
222       endMark = t;
223       if (endMark < startMark) {
224         double tm = startMark;
225         startMark = endMark;
226         endMark = tm;
227       }
228     } else {
229       startMark = Math.min(lastMark, t);
230       endMark = Math.max(lastMark, t);
231     }
232     lastMark = t;
233     if (insetWavePanel != null) {
234       insetWavePanel.setMarks(startMark, endMark);
235     }
236     repaint();
237   }

389   /**
390    * Process mouse position.
391    * @param x mouse x position
392    * @param y mouse y position
393    */
394   private void processMousePosition(int x, int y) {
395     if (heliData == null) {
396       return;
397     }
398 
399     String status = null;
400 
401     boolean wp = false;
402     if (insetWavePanel != null) {
403       Point loc = insetWavePanel.getLocation();
404       wp = insetWavePanel.processMousePosition(x - loc.x, y - loc.y);
405     }
406 
407     if (!wp) {
408       if (status == null) {
409         if (!(x < heliRenderer.getGraphX() || y < heliRenderer.getGraphY()
410             || x > heliRenderer.getGraphX() + heliRenderer.getGraphWidth() - 1
411             || y > heliRenderer.getGraphY() + heliRenderer.getGraphHeight() - 1)) {
412 
413           double j2k = getMouseJ2K(x, y);
414           status = StatusTextArea.getTimeString(j2k, swarmConfig.getTimeZone(settings.channel));
415         }
416       }
417 
418       if (status == null) {
419         status = " ";
420       }
421 
422       if (status != null) {
423         parent.setStatus(status);
424       }
425     }
426   }

428   /**
429    * Get time at point.
430    * @param mx mouse x position
431    * @param my mouse y position
432    * @return time in J2K
433    */
434   public double getMouseJ2K(int mx, int my) {
435     double j2k = 0;
436     if (translation != null) {
437       j2k = translation[4];
438       j2k += (mx - translation[0]) * translation[7];
439       j2k += getHelicorderRow(my) * translation[6];
440     }
441     return j2k;
442   }
443 
444   public int getHelicorderRow(int my) {
445     return (int) Math.floor((my - translation[3]) / translation[2]);
446   }

460   /**
461    * Move selected wave window.
462    * 
463    * @param offset window lengths to move
464    */
465   public void moveInset(int offset) {
466     if (insetWavePanel == null) {
467       return;
468     }
469 
470     double st = insetWavePanel.getStartTime();
471     double et = insetWavePanel.getEndTime();
472     double dt = et - st;
473     double newStartTime = st + dt * offset;
474     double newEndTime = et + dt * offset;
475 
476     int firstRow = heliRenderer.getRow(newStartTime);
477     int lastRow = heliRenderer.getRow(newEndTime);
478     if (lastRow < 0 || firstRow > heliRenderer.getNumRows() - 1) {
479       return;
480     }
481 
482     loadInsetWave(newStartTime, newEndTime);
483   }

485   /**
486    * Create wave inset.
487    * @param j2k time
488    * @param mx mouse x position
489    * @param my mouse y position
490    */
491   public void createWaveInset(final double j2k, final int mx, final int my) {
492     if (working) {
493       return;
494     }
495 
496     insetY = my;
497 
498     if (insetWavePanel == null) {
499       insetWavePanel = new WaveViewPanel(parent.getWaveViewSettings());
500       insetWavePanel.addListener(new WaveViewPanelAdapter() {
501         public void waveClosed(WaveViewPanel src) {
502           removeWaveInset();
503         }
504 
505         public void waveTimePressed(WaveViewPanel src, MouseEvent e, double j2k) {
506           if (swarmConfig.durationEnabled && SwingUtilities.isLeftMouseButton(e)) {
507             markTime(j2k);   
508           }
509           insetWavePanel.processMousePosition(e.getX(), e.getY());
510         }
511       });
512     }
513 
514     // insetWavePanel.setHelicorderPanel(this);
515     insetWavePanel.setMarks(startMark, endMark);
516     insetWavePanel.setChannel(settings.channel);
517     insetWavePanel.setDataSource(parent.getDataSource());
518     insetWavePanel.setStatusText(parent.getStatusText());
519 
520     Dimension d = getSize();
521     insetHeight = getHeight() / 4;
522     int height = insetHeight;
523     int row = heliRenderer.getRow(j2k);
524 
525     insetWavePanel.setSize(d.width + 2, height);
526     double zoomOffset = parent.getHelicorderViewerSettings().waveZoomOffset;
527     int rowSpan = (int) Math.ceil(2 * zoomOffset / heliRenderer.getTimeChunk());
528     if (insetY - heliRenderer.getRowHeight() > height + translation[GRAPH_Y]) {
529       int y = (int) Math.ceil((row - rowSpan) * translation[ROW_HEIGHT] + translation[GRAPH_Y]);
530       insetWavePanel.setLocation(-1, y - height);
531     } else {
532       int y = (int) Math.ceil((row + 1 + rowSpan) * translation[ROW_HEIGHT] + translation[GRAPH_Y]);
533       insetWavePanel.setLocation(-1, y);
534     }
535 
536     insetWavePanel.setAllowClose(true);
537     insetWavePanel.setWorking(true);
538 
539     loadInsetWave(j2k - zoomOffset, j2k + zoomOffset);
540     parent.setInsetButtonsEnabled(true);
541     this.add(insetWavePanel);
542     repaint();
543   }

545   private void loadInsetWave(final double st, final double et) {
546     fireInsetCreated(st, et);
547     final SwingWorker worker = new SwingWorker() {
548       private Wave sw;
549 
550       public Object construct() {
551         parent.getThrobber().increment();
552         working = true;
553         sw = parent.getWave(st, et);
554         return null;
555       }
556 
557       public void finished() {
558         parent.getThrobber().decrement();
559         working = false;
560         if (insetWavePanel != null) {
561           insetWavePanel.setWave(sw, st, et);
562           insetWavePanel.setWorking(false);
563         }
564         repaint();
565       }
566     };
567     worker.start();
568   }

574    * @param time2 end time

576   public void setHelicorder(HelicorderData d, double time1, double time2) {
577     heliData = d;
578     if (heliData != null) {
579       startTime = time1;
580       endTime = time2;
581       heliRenderer.setData(heliData);
582       heliRenderer.setTimeChunk(settings.timeChunk);
583       heliRenderer.setTimeZone(swarmConfig.getTimeZone(settings.channel));
584       heliRenderer.setForceCenter(settings.forceCenter);
585       heliRenderer.setClipBars(settings.clipBars);
586       heliRenderer.setShowClip(settings.showClip);
587       heliRenderer.setAlertClip(settings.alertClip);
588       heliRenderer.setClipWav("clip.wav");
589       heliRenderer.setClipAlertTimeout(settings.alertClipTimeout);
590       mean = heliData.getMeanMax();
591       bias = heliData.getBias();
592       if (bias != mean) {
593         mean = Math.abs(bias - mean);
594       }
595       heliRenderer.setClipValue(settings.clipValue);
596 
597     }
598   }

622   private void createImage() {
623     if (heliData == null) {
624       return;
625     }
626 
627     Dimension d = this.getSize();
628     if (d.width <= 0 || d.height <= 0) {
629       return;
630     }
631 
632     bufferImage = new BufferedImage(d.width, d.height, BufferedImage.TYPE_4BYTE_ABGR);
633 
634     Graphics2D ig = (Graphics2D) bufferImage.getGraphics();
635     plot.setSize(d);
636 
637     double offset = 0;
638     double multiplier = 1;
639     Metadata md = swarmConfig.getMetadata(settings.channel);
640     if (md != null) {
641       offset = md.getOffset();
642       multiplier = md.getMultiplier();
643     }
644 
645     if (minimal) {
646       heliRenderer.setLocation(X_OFFSET / 2, Y_OFFSET, d.width - X_OFFSET - 5,
647           d.height - Y_OFFSET - BOTTOM_HEIGHT / 2);
648     } else {
649       heliRenderer.setLocation(X_OFFSET, Y_OFFSET, d.width - X_OFFSET - RIGHT_WIDTH,
650           d.height - Y_OFFSET - BOTTOM_HEIGHT);
651     }
652 
653     if (settings.autoScale) {
654       settings.barRange = (int) (mean * settings.barMult);
655       settings.clipValue = (int) (mean * settings.clipBars);
656       heliRenderer.setHelicorderExtents(startTime, endTime, -1 * Math.abs(settings.barRange),
657           Math.abs(settings.barRange));
658 
659     } else {
660       heliRenderer.setHelicorderExtents(startTime, endTime,
661           -1 * Math.abs((settings.barRange - offset) / multiplier),
662           Math.abs((settings.barRange - offset) / multiplier));
663     }
664 
665     heliRenderer.setTimeZone(swarmConfig.getTimeZone(settings.channel));
666     heliRenderer.setClipValue(settings.clipValue);
667     if (minimal) {
668       // System.out.println("minimal");
669       // heliRenderer.createMinimumAxis();
670       heliRenderer.setFrameDecorator(new SmallDecorator());
671     } else {
672       heliRenderer.createDefaultAxis();
673     }
674 
675     if (md == null || md.getAlias() == null) {
676       heliRenderer.setChannel(settings.channel);
677     } else {
678       heliRenderer.setChannel(md.getAlias());
679     }
680 
681     translation = heliRenderer.getTranslationInfo(false);
682     heliRenderer.setLargeChannelDisplay(fullScreen);
683 
684     try {
685       plot.render(ig);
686     } catch (PlotException e) {
687       e.printStackTrace();
688     }
689   }

691   class SmallDecorator extends FrameDecorator {
692     public void decorate(FrameRenderer fr) {
693       AxisRenderer axis = new AxisRenderer(fr);
694       axis.createDefault();
695       fr.setAxis(axis);
696 
697       int minutes = (int) Math.round(settings.timeChunk / 60.0);
698       int majorTicks = minutes;
699       if (minutes > 30 && minutes < 180) {
700         majorTicks = minutes / 5;
701       } else if (minutes >= 180 && minutes < 360) {
702         majorTicks = minutes / 10;
703       } else if (minutes >= 360) {
704         majorTicks = minutes / 20;
705       }
706       double[] mjt = SmartTick.intervalTick(0, settings.timeChunk, majorTicks);
707 
708       axis.createBottomTicks(null, mjt);
709       axis.createTopTicks(null, mjt);
710       axis.createVerticalGridLines(mjt);
711 
712       int bc = (settings.timeChunk / 5) + 2;
713       String[] btl = new String[bc];
714       double[] btlv = new double[bc];
715       btl[0] = "+";
716       btlv[0] = 30;
717       for (int i = 0, j = 1; i < mjt.length; i++) {
718         long m = Math.round(mjt[i] / 60.0);
719         if (m % 5 == 0) {
720           btl[j] = Long.toString(m);
721           btlv[j++] = mjt[i];
722         }
723       }
724       axis.createBottomTickLabels(btlv, btl);
725 
726       HelicorderRenderer hr = (HelicorderRenderer) fr;
727       int numRows = hr.getNumRows();
728       double[] labelPosLr = new double[numRows];
729       String[] leftLabelText = new String[numRows];
730       String[] rightLabelText = new String[numRows];
731       TimeZone timeZone = swarmConfig.getTimeZone(settings.channel);
732 
733       DateFormat localTimeFormat = new SimpleDateFormat("HH:mm");
734       localTimeFormat.setTimeZone(timeZone);
735 
736       DateFormat localDayFormat = new SimpleDateFormat("MM-dd");
737       localDayFormat.setTimeZone(timeZone);
738 
739       double pixelsPast = 0;
740       double pixelsPerRow = fr.getGraphHeight() / numRows;
741       String lastLocalDay = "";
742       for (int i = numRows - 1; i >= 0; i--) {
743         pixelsPast += pixelsPerRow;
744         labelPosLr[i] = i + 0.5;
745 
746         String localTime = localTimeFormat
747             .format(J2kSec.asDate(hr.getHelicorderMaxX() - (i + 1) * settings.timeChunk));
748         leftLabelText[i] = null;
749         if (pixelsPast > 20) {
750           leftLabelText[i] = localTime;
751           pixelsPast = 0;
752         }
753 
754         Date dtz = J2kSec.asDate(hr.getHelicorderMaxX() - (i + 1) * settings.timeChunk);
755         String localDay =
756             localDayFormat.format(new Date(dtz.getTime() + settings.timeChunk * 1000));
757         if (!localDay.equals(lastLocalDay)) {
758           rightLabelText[i] = localDay;
759           lastLocalDay = localDay;
760         }
761       }
762 
763       axis.createLeftTickLabels(labelPosLr, leftLabelText);
764       axis.createRightTickLabels(labelPosLr, rightLabelText);
765 
766       boolean dst = timeZone.inDaylightTime(J2kSec.asDate(hr.getViewEndTime()));
767       String timeZoneName = timeZone.getDisplayName(dst, TimeZone.SHORT);
768       TextRenderer tr =
769           new TextRenderer(3, fr.getGraphY() + fr.getGraphHeight() + 14, timeZoneName);
770       tr.font = Font.decode("dialog-PLAIN-9");
771       tr.antiAlias = false;
772       axis.addPostRenderer(tr);
773 
774       double[] hg = new double[numRows - 1];
775       for (int i = 0; i < numRows - 1; i++) {
776         hg[i] = i + 1.0;
777       }
778 
779       axis.createHorizontalGridLines(hg);
780 
781       axis.setBackgroundColor(Color.white);
782     }

793   private void drawMark(Graphics2D g2, double t, Color color) {
794     if (Double.isNaN(t)) {
795       return;
796     }
797 
798     int x = (int) (heliRenderer.helicorderGetXPixel(t));
799     int row = heliRenderer.getRow(t);
800     int y = (int) Math.ceil(row * translation[2] + translation[3]);
801     g2.setColor(color);
802     g2.draw(new Line2D.Double(x, y, x, y + translation[2]));
803 
804     GeneralPath gp = new GeneralPath();
805     gp.moveTo(x, y);
806     gp.lineTo((float) x - 4, (float) y - 6);
807     gp.lineTo((float) x + 4, (float) y - 6);
808     gp.closePath();
809     g2.setColor(Color.GREEN);
810     g2.fill(gp);
811     g2.setColor(DARK_GREEN);
812     g2.draw(gp);
813 
814     gp.reset();
815     gp.moveTo(x, (float) (y + translation[2]));
816     gp.lineTo((float) x - 4, (float) (y + 6 + translation[2]));
817     gp.lineTo((float) x + 4, (float) (y + 6 + translation[2]));
818     gp.closePath();
819     gp.closePath();
820     g2.setColor(Color.GREEN);
821     g2.fill(gp);
822     g2.setColor(DARK_GREEN);
823     g2.draw(gp);
824   }

828   /**
829    * @see javax.swing.JComponent#paint(java.awt.Graphics)
830    */
831   public void paint(Graphics g) {
832     Graphics2D g2 = (Graphics2D) g;
833     Dimension d = this.getSize();
834     if (heliData == null) {
835       if (!parent.isWorking()) {
836         parent.setStatus("The server returned no helicorder data.");
837       }
838     } else if (displayImage != null) {
839       g2.drawImage(displayImage, 0, 0, null);
840     }
841 
842     drawMark(g2, startMark, DARK_GREEN);
843     drawMark(g2, endMark, DARK_GREEN);
844 
845     if (insetWavePanel != null) {
846       // find out where time highlight will be, possibly reposition the
847       // insetWavePanel
848       double t1 = insetWavePanel.getStartTime();
849       double t2 = insetWavePanel.getEndTime();
850       double dt = t2 - t1;
851       double spanCenter = (t2 - t1) / 2 + t1;
852       int row = heliRenderer.getRow(spanCenter);
853 
854       int rowSpan = (int) Math.ceil(dt / heliRenderer.getMaxX()) + 1;
855       double rowHeight = translation[ROW_HEIGHT];
856       double graphY = translation[GRAPH_Y];
857 
858       if (resized) {
859         insetWavePanel.setSize(d.width, insetHeight);
860         insetWavePanel.createImage();
861       }
862 
863       int panelY;
864       if ((row - rowSpan) * rowHeight > insetHeight + graphY) {
865         panelY = (int) Math.ceil((row - rowSpan) * rowHeight + graphY - insetHeight);
866       } else {
867         panelY = (int) Math.ceil((row + rowSpan) * rowHeight + graphY);
868       }
869 
870       if (panelY != insetWavePanel.getLocation().y) {
871         insetWavePanel.setLocation(0, panelY);
872       }
873 
874       // now it's safe to draw the waveInsetPanel
875       Point p = insetWavePanel.getLocation();
876       g2.translate(p.x, p.y);
877       insetWavePanel.paint(g2);
878       Dimension wvd = insetWavePanel.getSize();
879       g.setColor(Color.gray);
880       g.drawRect(0, 0, wvd.width - 1, wvd.height);
881       g2.translate(-p.x, -p.y);
882 
883       /*
884        * Not sure what below is meant to address. I removed the check because negative row offsets
885        * were necessary to handle multi-row highlights. If there's a problem with wave loading, find
886        * a more direct fix. --tjp
887        */
888       // fixes bug where highlight was being drawn before wave loaded
889       // if (row < 0)
890       // return;
891 
892       // finally, draw the highlight.
893       Paint pnt = g2.getPaint();
894       g2.setPaint(new Color(255, 255, 0, 128));
895       Rectangle2D.Double rect = new Rectangle2D.Double();
896 
897       int zoomTimeSpan = parent.getHelicorderViewerSettings().waveZoomOffset * 2;
898       double zoomPixelSpan = 1.0 / translation[PIXEL_TIME_SPAN] * zoomTimeSpan;
899 
900       // highlight left
901       int highlightStart = (int) heliRenderer.helicorderGetXPixel(spanCenter);
902       int highlightSpan = (int) (zoomPixelSpan / 2);
903       int rowOffset = 0;
904       while (highlightSpan > 0 && row - rowOffset >= 0) {
905         int width = Math.min(highlightSpan, (int) (highlightStart - translation[GRAPH_LEFT]));
906 
907         if (row - rowOffset < heliRenderer.getNumRows()) {
908           rect.setRect(highlightStart - width,
909               (int) Math.ceil((row - rowOffset) * translation[ROW_HEIGHT] + translation[GRAPH_Y]),
910               width, (int) Math.ceil(translation[ROW_HEIGHT]));
911           g2.fill(rect);
912         }
913         highlightSpan -= width;
914         highlightStart = (int) translation[GRAPH_RIGHT];
915         rowOffset++;
916 
917       }
918 
919       // highlight right
920       highlightStart = (int) heliRenderer.helicorderGetXPixel(spanCenter);
921       highlightSpan = (int) (zoomPixelSpan / 2);
922       rowOffset = 0;
923       while (highlightSpan > 0 && row + rowOffset < heliRenderer.getNumRows()) {
924         int width = Math.min(highlightSpan, (int) (translation[GRAPH_RIGHT] - highlightStart));
925 
926         if (row + rowOffset >= 0) {
927           rect.setRect(highlightStart,
928               (int) Math.ceil((row + rowOffset) * translation[ROW_HEIGHT] + translation[GRAPH_Y]),
929               width, (int) Math.ceil(translation[ROW_HEIGHT]));
930           g2.fill(rect);
931         }
932 
933         highlightSpan -= width;
934         highlightStart = (int) translation[GRAPH_LEFT];
935         rowOffset++;
936       }
937 
938       g2.setPaint(pnt);
939     }
940 
941     resized = false;
942   }
