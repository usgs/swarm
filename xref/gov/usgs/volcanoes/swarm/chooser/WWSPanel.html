17  public class WWSPanel extends DataSourcePanel {
18    private JTextField wwsHost;
19    private JTextField wwsPort;
20    private JTextField wwsTimeout;
21    private JCheckBox wwsCompress;
22  
23    public WWSPanel() {
24      super("wws", "Winston Wave Server");
25    }
26  
27    private void createFields() {
28      wwsHost = new JTextField();
29      wwsPort = new JTextField();
30      wwsTimeout = new JTextField();
31      wwsCompress = new JCheckBox();
32      resetSource(source);
33    }
34  
35    public void resetSource(String source) {
36      this.source = source;
37      String h = "";
38      String p = "16022";
39      String t = "15.0";
40      boolean wscomp = true;
41      if (source != null && source.indexOf(";wws:") != -1) {
42        String[] ss = source.substring(source.indexOf(";wws:") + 5).split(":");
43        h = ss[0];
44        p = ss[1];
45        t = String.format("%.1f", Integer.parseInt(ss[2]) / 1000.0);
46        wscomp = ss[3].equals("1");
47      }
48      wwsHost.setText(h);
49      wwsPort.setText(p);
50      wwsTimeout.setText(t);
51      wwsCompress.setSelected(wscomp);
52    }
53    @Override
54    protected void createPanel() {
55      createFields();
56      FormLayout layout = new FormLayout("right:max(20dlu;pref), 3dlu, 40dlu, 0dlu, 126dlu", "");
57  
58      DefaultFormBuilder builder = new DefaultFormBuilder(layout);
59      builder.setDefaultDialogBorder();
60      builder.append(new JLabel("Use this data source to connect to a Winston Wave Server (WWS)."),
61          5);
62      builder.nextLine();
63      builder.appendSeparator();
64      builder.append("IP Address or Host Name:");
65      builder.append(wwsHost, 3);
66      builder.nextLine();
67      builder.append("Port:");
68      builder.append(wwsPort);
69      builder.append(" Winston default: 16022");
70      builder.nextLine();
71  
72      builder.append("Timeout:");
73      builder.append(wwsTimeout);
74      builder.append(" seconds");
75      builder.nextLine();
76  
77      builder.append("Use Compression:");
78      builder.append(wwsCompress);
79      builder.nextLine();
80  
81      panel = builder.getPanel();
82    }
83  
84    @Override
85    public boolean allowOK(boolean edit) {
86      String host = wwsHost.getText();
87      String message = null;
88  
89      if (host == null || host.length() == 0 || host.indexOf(';') != -1 || host.indexOf(':') != -1)
90        message = "There is an error with the WWS IP address or host name.";
91      
92      int ip = StringUtils.stringToInt(wwsPort.getText(), -1);
93      if (ip < 0 || ip > 65535)
94        message = "There is an error with the WWS port.";
95  
96      double to = StringUtils.stringToDouble(wwsTimeout.getText(), -1);
97      if (to <= 0)
98        message = "There is an error with the WWS time out (must be > 0).";
99      
100     if (message != null) {
101       JOptionPane.showMessageDialog(applicationFrame, message, "Error", JOptionPane.ERROR_MESSAGE);
102       return false;
103     } else
104       return true;
105   }
106 
107   @Override
108   public String wasOK() {
109     int timeout = (int) (Double.parseDouble(wwsTimeout.getText()) * 1000);
110     String result = String.format("wws:%s:%s:%d:%s", wwsHost.getText(), wwsPort.getText(), timeout,
111         (wwsCompress.isSelected() ? "1" : "0"));
112     return result;
113   }
