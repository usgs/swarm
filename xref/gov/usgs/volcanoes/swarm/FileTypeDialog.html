23  public class FileTypeDialog extends SwarmDialog {
24      private static final long serialVersionUID = 1L;
25      private static final JFrame applicationFrame = Swarm.getApplicationFrame();
26      private JLabel filename;
27      private JList fileTypes;
28      private JCheckBox assumeSame;
29      private boolean cancelled = true;
30      private boolean opened = false;
31  
32      public FileTypeDialog() {
33          super(applicationFrame, "Select File Type", true);
34          setSizeAndLocation();
35      }
36  
37      public void setFilename(String fn) {
38          filename.setText(fn);
39      }
40  
41      protected void createUI() {
42          super.createUI();
43          filename = new JLabel();
44          filename.setFont(Font.decode("dialog-BOLD-12"));
45          filename.setBorder(BorderFactory.createEmptyBorder(0, 0, 4, 0));
46          
47          fileTypes = new JList(FileType.getKnownTypes());
48          fileTypes.addMouseListener(new MouseAdapter() {
49              public void mouseClicked(MouseEvent e) {
50                  if (e.getClickCount() == 2) {
51                      if (fileTypes.getSelectedIndex() != -1)
52                          okButton.doClick();
53                  }
54              }
55          });
56          fileTypes.setSelectedValue(FileType.UNKNOWN, true);
57          assumeSame = new JCheckBox("Assume all unknown files are of this type", false);
58          JPanel panel = new JPanel(new BorderLayout());
59          panel.setBorder(BorderFactory.createEmptyBorder(5, 9, 5, 9));
60          panel.setPreferredSize(new Dimension(300, 200));
61          JPanel labelPanel = new JPanel(new GridLayout(3, 1));
62          labelPanel.add(new JLabel("Select file type: "));
63          labelPanel.add(filename);
64          labelPanel.add(new JLabel("Choose 'Cancel' to skip this file or select file type:"));
65          panel.add(labelPanel, BorderLayout.NORTH);
66          panel.add(new JScrollPane(fileTypes), BorderLayout.CENTER);
67          panel.add(assumeSame, BorderLayout.SOUTH);
68          mainPanel.add(panel, BorderLayout.CENTER);
69      }
70  
71      public boolean isAssumeSame() {
72          return assumeSame.isSelected();
73      }
74  
75      public FileType getFileType() {
76          return (FileType) fileTypes.getSelectedValue();
77      }
78  
79      public void wasOK() {
80          cancelled = false;
81      }
82  
83      public void wasCancelled() {
84          cancelled = true;
85          opened = false;
86      }
87  
88      public void setVisible(boolean b) {
89          if (b)
90              opened = true;
91          super.setVisible(b);
92      }
93      
94      public boolean isOpen() {
95          return opened;
96      }
97      
98      public boolean isCancelled() {
99          return cancelled;
100     }
101 }
