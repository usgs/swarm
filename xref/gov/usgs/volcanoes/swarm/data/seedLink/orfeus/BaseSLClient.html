96       *  Creates a new instance of BaseSLClient with the specified logging object.
97       *
98       * @param sllog logging object to handle messages.
99       *
100      */
101     public BaseSLClient(SLLog sllog) {
102     	final StringBuilder sb = new StringBuilder();
103         for (int n = 0; n < SLClient.BANNER.length; n++) {
104             if (sb.length() != 0)
105             {
106             	sb.append('\n');
107             }
108             sb.append(SLClient.BANNER[n]);
109         }
110         sllog.log(false, 1, sb.toString());
111 
112         this.sllog = sllog;
113 
114         slconn = new BaseSLConnection(sllog);
115 
116     }

118     /**
119      *
120      * Parses the commmand line arguments.
121      *
122      * @param args the main method arguments.
123      *
124      *
125      * @return -1 on error, 1 if version or help argument found,  0 otherwise.
126      *
127      */
128     public int parseCmdLineArgs(String[] args) {
129 
130         if (args.length < 1) {
131             printUsage(false);
132             return (1);
133         }
134 
135         int optind = 0;
136 
137         while (optind < args.length) {
138 
139             if (args[optind].equals("-V")) {
140                 System.err.println(SLClient.VERSION_INFO);
141                 return (1);
142             } else if (args[optind].equals("-h")) {
143                 printUsage(false);
144                 return (1);
145             } else if (args[optind].startsWith("-v")) {
146                 verbose += args[optind].length() - 1;
147                 if (sllog instanceof BaseSLLog)
148                 {
149                 	((BaseSLLog)sllog).setVerbosity(verbose);
150                 }
151             } else if (args[optind].equals("-p")) {
152                 ppackets = true;
153             } else if (args[optind].equals("-nt")) {
154                 slconn.setNetTimout(Integer.parseInt(args[++optind]));
155             } else if (args[optind].equals("-nd")) {
156                 slconn.setNetDelay(Integer.parseInt(args[++optind]));
157             } else if (args[optind].equals("-k")) {
158                 slconn.setKeepAlive(Integer.parseInt(args[++optind]));
159             } else if (args[optind].equals("-l")) {
160                 streamfile = args[++optind];
161             } else if (args[optind].equals("-s")) {
162                 selectors = args[++optind];
163             } else if (args[optind].equals("-S")) {
164                 multiselect = args[++optind];
165             } else if (args[optind].equals("-x")) {
166                 statefile = args[++optind];
167             } else if (args[optind].equals("-t")) {
168                 begin_time = args[++optind];
169             } else if (args[optind].equals("-e")) {
170                 end_time = args[++optind];
171             } else if (args[optind].equals("-i")) {
172                 infolevel = args[++optind];
173             } else if (args[optind].startsWith("-")) {
174                 System.err.println("Unknown option: " + args[optind]);
175                 return (-1);
176             } else if (slconn.getSLAddress() == null) {
177                 slconn.setSLAddress(args[optind]);
178             } else {
179                 System.err.println("Unknown option: " + args[optind]);
180                 return (-1);
181             }
182             optind++;
183 
184         }
185 
186         return (0);
187 
188     }

190     /**
191      *
192      * Initializes this SLCient.
193      *
194      * @exception SeedLinkException on error.
195      * @exception UnknownHostException if no IP address for the local host could be found.
196      *
197      */
198     public void init() throws UnknownHostException, SeedLinkException {
199 
200         // Make sure a server was specified
201         if (slconn.getSLAddress() == null) {
202             String message = "no SeedLink server specified";
203             throw (new SeedLinkException(message));
204         }
205 
206         // Report the program version
207         //sllog.log(false, 1, VERSION_INFO);
208 
209         // If verbosity is 2 or greater print detailed packet infor
210         if (verbose >= 2) {
211             ppackets = true;
212         }
213 
214         // If no host is given for the SeedLink server, add 'localhost'
215         if (slconn.getSLAddress().startsWith(":")) {
216             slconn.setSLAddress(InetAddress.getLocalHost().toString() + slconn.getSLAddress());
217         }
218 
219         // Load the stream list from a file if specified
220         if (streamfile != null) {
221             slconn.readStreamList(streamfile, selectors);
222         }
223 
224         // Parse the 'multiselect' string following '-S'
225         if (multiselect != null) {
226             slconn.parseStreamlist(multiselect, selectors);
227         } else if (streamfile == null) // No 'streams' array, assuming uni-station mode
228         {
229             slconn.setUniParams(selectors, -1, null);
230         }
231 
232         // Attempt to recover sequence numbers from state file
233         if (statefile != null) {
234             slconn.setStateFile(statefile);
235         } else {
236             // Set begin time for read start in past
237             // 20050415 AJL added to support continuous data transfer from a time in the past
238             if (begin_time != null) {
239                 slconn.setBeginTime(begin_time);
240             }
241             // Set end time for for reading windowed data
242             // 20071204 AJL added
243             if (end_time != null) {
244                 slconn.setEndTime(end_time);
245             }
246         }
247 
248         //slconn.lastpkttime = true;
249 
250     }

256      */
257     public void run() {
258     	try {
259 		   oldrun();
260     	} catch (Exception ex) {
261     		sllog.log(true, 0, ex.getMessage());
262     	}
263         // Close the BaseSLConnection
264         slconn.close();
265     }

267     /**
268      *
269      * Start this SLCient.
270      *
271      * @exception SeedLinkException on error.
272      * @exception IOException if an I/O error occurs.
273      *
274      */
275     protected void oldrun() throws Exception {
276 
277         if (infolevel != null) {
278             slconn.requestInfo(infolevel);
279         }
280 
281         // Loop with the connection manager
282         SLPacket slpack = null;
283         int count = 1;
284         while ((slpack = slconn.collect()) != null) {
285 
286             if (slpack == SLPacket.SLTERMINATE) {
287                 break;
288             }
289 
290             slpack.getType(); // ensure the blockette is created if needed
291             // reset the volume counter
292         	BlocketteDecoratorFactory.reset();
293 
294             try {
295                 // do something with packet
296                 boolean terminate = packetHandler(count, slpack);
297                 if (terminate) {
298                     break;
299                 }
300 
301             } catch (SeedLinkException sle) {
302                 sllog.log(true, 0, CLASS_NAME + ": " + sle);
303             }
304 
305             // 20081127 AJL - test modification to prevent "Error: out of java heap space" problem identified by pwiejacz@igf.edu.pl
306             if (count >= Integer.MAX_VALUE) {
307                 count = 1;
308                 sllog.log(false, 1, CLASS_NAME + ": Packet count reset to 1");
309             } else {
310                 count++;
311             }
312         }
313     }

315     /**
316      *
317      * Method that processes each packet received from the BaseSLConnection.
318      * This mehod should be overridded when subclassing BaseSLClient.
319      *
320      * @param count the packet to process.
321      * @param slpack the packet to process.
322      *
323      * @return true if connection to SeedLink server should be closed and session terminated, false otherwise.
324      *
325      * @exception implementation dependent
326      *
327      */
328     public boolean packetHandler(int count, SLPacket slpack) throws Exception {
329 
330         // check if not a complete packet
331         if (slpack == null || slpack == SLPacket.SLNOPACKET || slpack == SLPacket.SLERROR) {
332             return (false);
333         }
334 
335         // get basic packet info
336         int seqnum = slpack.getSequenceNumber();
337         int type = slpack.getType();
338 
339         // process INFO packets here
340         if (type == SLPacket.TYPE_SLINF) {
341             //System.out.println("Unterminated INFO packet: [" + (new String(slpack.msrecord, 0, 20)) + "]");
342             return (false);
343         }
344         if (type == SLPacket.TYPE_SLINFT) {
345             //System.out.println("Terminated INFO packet: [" + (new String(slpack.msrecord, 0, 20)) + "]");
346             sllog.log(false, 0, "Complete INFO:\n" + slconn.getInfoString());
347             if (infolevel != null) {
348                 return (true);
349             } else {
350                 return (false);
351             }
352         }
353 
354 
355         // can send an in-line INFO request here
356         if (count % 100 == 0) {
357             String infostr = "ID";
358             slconn.requestInfo(infostr);
359         }
360 
361 
362         // if here, must be a blockette
363 
364         sllog.log(false, 0, CLASS_NAME + ": packet seqnum: " + seqnum + ": packet type: " + type);
365 
366         if (!ppackets) {
367             return (false);
368         }
369 
370         Blockette blockette = slpack.getBlockette();
371 
372         sllog.log(false, 0, CLASS_NAME + ": blockette type: " + blockette.getType() + ": blockette: " + blockette);
373 
374         Waveform waveform = blockette.getWaveform();
375         if (waveform != null) {
376             sllog.log(false, 0, CLASS_NAME + ": blockette contains a waveform of length: " + waveform.getNumSamples());
377         } else {
378             sllog.log(false, 0, CLASS_NAME + ": blockette contains no waveform");
379         }
380 
381         return (false);
382 
383     }

386      *
387      * Prints the usage message for this class.
388      *
389      */
390     public void printUsage(boolean concise) {
391         System.err.println("\nUsage: java [-cp classpath] " + PACKAGE + "." + CLASS_NAME + " [options] <[host]:port>\n");
392         if (concise) {
393             System.err.println("Use '-h' for detailed help");
394             return;
395         }
396         System.err.println(" ## General program options ##\n"
397                 + " -V             report program version\n"
398                 + " -h             show this usage message\n"
399                 + " -v             be more verbose, multiple flags can be used\n"
400                 + " -p             print details of data packets\n\n"
401                 + " -nd delay      network re-connect delay (seconds), default 30\n"
402                 + " -nt timeout    network timeout (seconds), re-establish connection if no\n"
403                 + "                  data/keepalives are received in this time, default 600\n"
404                 + " -k interval    send keepalive (heartbeat) packets this often (seconds)\n"
405                 + " -x statefile   save/restore stream state information to this file\n"
406                 + " -t begintime   sets a beginning time for the initiation of data transmission (year,month,day,hour,minute,second)\n"
407                 + " -e endtime     sets an end time for windowed data transmission  (year,month,day,hour,minute,second)\n"
408                 + " -i infolevel   request this INFO level, write response to std out, and exit \n"
409                 + "                  infolevel is one of: ID, STATIONS, STREAMS, GAPS, CONNECTIONS, ALL \n"
410                 + "\n"
411                 + " ## Data stream selection ##\n"
412                 + " -l listfile    read a stream list from this file for multi-station mode\n"
413                 + " -s selectors   selectors for uni-station or default for multi-station\n"
414                 + " -S streams     select streams for multi-station (requires SeedLink >= 2.5)\n"
415                 + "   'streams' = 'stream1[:selectors1],stream2[:selectors2],...'\n"
416                 + "        'stream' is in NET_STA format, for example:\n"
417                 + "        -S \"IU_KONO:BHE BHN,GE_WLF,MN_AQU:HH?.D\"\n"
418                 + "\n"
419                 + " <[host]:port>  Address of the SeedLink server in host:port format\n"
420                 + "                  if host is omitted (i.e. ':18000'), localhost is assumed\n\n");
421 
422     }

424     /**
425      *
426      * main method
427      *
428      */
429     public static void main(String[] args) {
430 
431         BaseSLClient slClient = null;
432 
433         try {
434             slClient = new BaseSLClient();
435             int rval = slClient.parseCmdLineArgs(args);
436             if (rval != 0) {
437                 System.exit(rval);
438             }
439             slClient.init();
440             slClient.run();
441         } catch (SeedLinkException sle) {
442             slClient.sllog.log(true, 0, sle.getMessage());
443         } catch (Exception e) {
444             System.err.println("ERROR: " + e.getMessage());
445             e.printStackTrace();
446         }
447     }
