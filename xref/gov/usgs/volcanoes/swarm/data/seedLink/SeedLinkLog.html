1   package gov.usgs.volcanoes.swarm.data.seedLink;
2   
3   import gov.usgs.util.Log;
4   import gov.usgs.volcanoes.swarm.data.seedLink.orfeus.BaseSLLog;
5   
6   import java.util.logging.Level;
7   import java.util.logging.Logger;
8   
9   /**
10   * SeedLink client log that uses Java logging.
11   * 
12   * @author Kevin Frechette (ISTI)
13   */
14  public class SeedLinkLog extends BaseSLLog
15  {
16  	/** Log prefix text. */
17  	private static final String LOG_PREFIX;
18  
19  	/** The logger. */
20  	public static final Logger logger;
21  	static
22  	{
23  		logger = Log.getLogger("gov.usgs.swarm.data.SeedLinkLog");
24  		logger.setLevel(Level.FINE);
25  		LOG_PREFIX = "SeedLink: ";
26  	}
27  
28  	/**
29  	 * Create the SeedLink client log.
30  	 */
31  	public SeedLinkLog()
32  	{
33  		setErrPrefix(LOG_PREFIX);
34  		setLogPrefix(LOG_PREFIX);
35  	}
36  
37  	/**
38  	 * 
39  	 * Logs a message in appropriate manner.
40  	 * 
41  	 * @param isError true if error message, false otherwise.
42  	 * @param verbosity verbosity level for this message.
43  	 * @param message message text.
44  	 */
45  	public void log(boolean isError, int verbosity, String message)
46  	{
47  		final Level level;
48  		if (isError) // error message
49  		{
50  			message = errPrefix + message;
51  			level = Level.INFO;
52  		}
53  		else
54  		{
55  			message = logPrefix + message;
56  			switch (verbosity)
57  			{
58  			case 0:
59  				level = Level.INFO;
60  				break;
61  			case 1:
62  				level = Level.FINE;
63  				break;
64  			case 2:
65  				level = Level.FINER;
66  				break;
67  			default:
68  				level = Level.FINEST;
69  				break;
70  			}
71  		}
72  		logger.log(level, message);
73  	}
74  
75  	public void setLevel(Level newLevel)
76  	{
77  		logger.setLevel(newLevel);
78  	}
79  }
