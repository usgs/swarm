72  	/**
73  	 * Create the query. All arguments must be provided.
74  	 * 
75  	 * @param network the network.
76  	 * @param station the station.
77  	 * @param location the location or empty for the empty location code.
78  	 * @param channel the channel.
79  	 * @return the query.
80  	 * @throws IOException if an I/O Exception occurs.
81  	 * @throws DataSelectException if data select error.
82  	 * @throws SeedFormatException if SEED format error.
83  	 * @see WebServiceUtils.EMPTY_LOC_CODE
84  	 */
85  	protected String createQuery(String network, String station,
86  			String location, String channel) 
87  	{
88  		if (location == null || location.trim().length() == 0)
89  		{
90  			location = WebServiceUtils.EMPTY_LOC_CODE;
91  		}
92  		String query = "net=" + network;
93  		query += "&sta=" + station;
94  		query += "&loc=" + location;
95  		query += "&cha=" + channel;
96  		return query;
97  	}

99  	/**
100 	 * Create the query. All arguments must be provided.
101 	 * 
102 	 * @param network the network.
103 	 * @param station the station.
104 	 * @param location the location or empty for the empty location code.
105 	 * @param channel the channel.
106 	 * @param begin the begin date.
107 	 * @param end the end date.
108 	 * @return the query.
109 	 * @see WebServiceUtils.EMPTY_LOC_CODE
110 	 * @see edu.sc.seis.seisFile.dataSelectWS.MSeedQueryReader#createQuery(java.lang
111 	 *      .String, java.lang.String, java.lang.String, java.lang.String,
112 	 *      java.util.Date, float)
113 	 */
114 	@Override
115 	public String createQuery(String network, String station, String location,
116 			String channel, Date begin, Date end)
117 	{
118 		String query;
119         query = createQuery(network, station, location, channel);
120 		SimpleDateFormat longFormat = new SimpleDateFormat(
121 				"yyyy-MM-dd'T'HH:mm:ss");
122 		longFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
123 		query += "&start=" + longFormat.format(begin);
124 		query += "&end=" + longFormat.format(end);
125 		return query;
126 	}

185 	/**
186 	 * Read the data records.
187 	 * 
188 	 * @param query the query.
189 	 * @param records the data record list or null if none.
190 	 * @return the data record list.
191 	 * @throws IOException if an I/O exception occurs.
192 	 * @throws DataSelectException if not OK repsonse code.
193 	 * @throws SeedFormatException if the SEED format is invalid.
194 	 */
195 	public List<DataRecord> read(String query, List<DataRecord> records)
196 			throws IOException, SeedFormatException
197 	{
198 		URL requestURL = new URL(urlBase + "?" + query);
199 		HttpURLConnection conn = (HttpURLConnection) requestURL
200 				.openConnection();
201 		if (timeoutMillis != 0)
202 		{
203 			conn.setReadTimeout(timeoutMillis);
204 		}
205 		conn.setRequestProperty("User-Agent", userAgent);
206 		conn.connect();
207 		if (conn.getResponseCode() != 200)
208 		{
209 			if (conn.getResponseCode() == 404)
210 			{
211 				WebServiceUtils.fine("reponse code 404, no data");
212 				return records;
213 			}
214 			else
215 			{
216 			    WebServiceUtils.fine(
217 						"Did not get an OK repsonse code (code="
218 								+ conn.getResponseCode() + ", url="
219 								+ requestURL + "\"");
220 			    return records;
221 			}
222 		}
223 		BufferedInputStream bif = new BufferedInputStream(conn.getInputStream());
224 		DataInputStream in = new DataInputStream(bif);
225 		while (true)
226 		{
227 			try
228 			{
229 				SeedRecord sr = SeedRecord.read(in);
230 				if (sr instanceof DataRecord)
231 				{
232 					if (processRecord((DataRecord) sr) && records != null)
233 					{
234 						records.add((DataRecord) sr);
235 					}
236 				}
237 				else
238 				{
239 					WebServiceUtils.warning("Not a data record, skipping..."
240 							+ sr.getControlHeader().getSequenceNum() + " "
241 							+ sr.getControlHeader().getTypeCode());
242 				}
243 			}
244 			catch (EOFException e)
245 			{
246 				// end of data?
247 				break;
248 			}
249 		}
250 		in.close();
251 		return records;
252 	}
