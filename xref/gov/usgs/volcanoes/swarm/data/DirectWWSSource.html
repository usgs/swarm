15  /**
16   * An implementation of <code>SeismicDataSource</code> that communicates
17   * directly with a Winston database. Essentially identical to
18   * DirectWinstonSource.
19   *
20   *
21   * @author Dan Cervelli
22   */
23  public class DirectWWSSource extends SeismicDataSource {
24    private String dbDriver;
25    private String dbURL;
26    private String dbPrefix;
27  
28    private WinstonDatabase winston;
29    private Data data;
30    private Channels stations;
31  
32    // explicit default constructor required for reflection
33    public DirectWWSSource() {}
34  
35    public void parse(String params) {
36      String[] ss = params.split("\\|");
37      dbDriver = ss[0];
38      dbURL = ss[1];
39      dbPrefix = ss[2];
40  
41      winston = new WinstonDatabase(dbDriver, dbURL, dbPrefix);
42      stations = new Channels(winston);
43      data = new Data(winston);
44    }
45  
46    public void close() {
47      winston.close();
48    }
49  
50    public synchronized Wave getWave(String station, double t1, double t2) {
51      CachedDataSource cache = CachedDataSource.getInstance();
52  
53      Wave sw = cache.getWave(station, t1, t2);
54      if (sw == null) {
55        try {
56          sw = data.getWave(station, t1, t2, 0);
57        } catch (UtilException e) {
58        }
59        if (sw != null && !sw.isData())
60          sw = null;
61        if (sw != null && sw.buffer != null && sw.buffer.length > 0)
62          cache.putWave(station, sw);
63      }
64      return sw;
65    }
66  
67    public synchronized List<String> getChannels() {
68      List<Channel> chs = stations.getChannels();
69      List<String> result = new ArrayList<String>();
70      for (Channel ch : chs)
71        result.add(ch.toString());
72      return result;
73    }
74  
75    public synchronized HelicorderData getHelicorder(String station, double t1, double t2,
76        GulperListener gl) {
77      CachedDataSource cache = CachedDataSource.getInstance();
78      HelicorderData hd = cache.getHelicorder(station, t1, t2, this);
79      if (hd == null) {
80        try {
81          hd = data.getHelicorderData(Scnl.parse(station), t1, t2, 0);
82        } catch (UtilException e) {
83        }
84        if (hd != null && hd.rows() != 0)
85          cache.putHelicorder(station, hd);
86        else
87          hd = null;
88      }
89      return hd;
90    }
91  
92    public String toConfigString() {
93      String typeString = DataSourceType.getShortName(this.getClass());
94      return String.format("%s;%s:%s|%s|%s", name, typeString, dbDriver, dbURL, dbPrefix);
95    }
96  }
