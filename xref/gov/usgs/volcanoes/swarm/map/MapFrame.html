451 	class CaptureActionListener implements ActionListener {
452 		public void actionPerformed(final ActionEvent e) {
453 			final JFileChooser chooser = FileChooser.getFileChooser();
454 			final File lastPath = new File(swarmConfig.lastPath);
455 			chooser.setCurrentDirectory(lastPath);
456 			chooser.setSelectedFile(new File("map.png"));
457 			chooser.setDialogTitle("Save Map Screen Capture");
458 			final int result = chooser.showSaveDialog(applicationFrame);
459 			File f = null;
460 			if (result == JFileChooser.APPROVE_OPTION) {
461 				f = chooser.getSelectedFile();
462 
463 				if (f.exists()) {
464 					final int choice = JOptionPane.showConfirmDialog(applicationFrame, "File exists, overwrite?",
465 							"Confirm", JOptionPane.YES_NO_OPTION);
466 					if (choice != JOptionPane.YES_OPTION)
467 						return;
468 				}
469 				swarmConfig.lastPath = f.getParent();
470 			}
471 			if (f == null)
472 				return;
473 
474 			final Insets i = mapPanel.getInsets();
475 			final BufferedImage image = new BufferedImage(mapPanel.getWidth() - i.left - i.right,
476 					mapPanel.getHeight() - i.top - i.bottom, BufferedImage.TYPE_4BYTE_ABGR);
477 			final Graphics g = image.getGraphics();
478 			g.translate(-i.left, -i.top);
479 			mapPanel.paint(g);
480 			try {
481 				final PngEncoderB png = new PngEncoderB(image, false, PngEncoder.FILTER_NONE, 7);
482 				final FileOutputStream out = new FileOutputStream(f);
483 				final byte[] bytes = png.pngEncode();
484 				out.write(bytes);
485 				out.close();
486 			} catch (final Exception ex) {
487 				ex.printStackTrace();
488 			}
489 		}
490 	}

518 	public void setView(final GeoRange gr) {
519 		final double lr1 = gr.getLonRange();
520 		final double lr2 = GeoRange.getLonRange(gr.getEast(), gr.getWest());
521 		if (lr2 < lr1)
522 			gr.flipEastWest();
523 
524 		mapPanel.setCenterAndScale(gr);
525 	}

574 	public void run() {
575 		while (true) {
576 			try {
577 				if (this.isVisible() && realtime) {
578 					final double end = J2kSec.now();
579 					final double start = end - MultiMonitor.SPANS[spanIndex];
580 					mapPanel.setTimes(start, end, false);
581 				}
582 
583 				Thread.sleep(refreshInterval);
584 			} catch (final Throwable e) {
585 				e.printStackTrace();
586 			}
587 		}
588 	}
