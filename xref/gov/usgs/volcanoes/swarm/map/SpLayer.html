42    /**
43     * Update list of S-P circles to draw.
44     */
45    private synchronized Vector<Sp> updateSpList() {
46      Vector<Sp> spList = new Vector<Sp>();
47      List<WaveViewPanel> waves = WaveClipboardFrame.getInstance().getWaves();
48      for (WaveViewPanel wvp : waves) {
49        if (wvp.getSettings().pickEnabled) {
50          PickData pickData = wvp.getPickData();
51          if (pickData.isPlot() && pickData.getPick(PickData.P) != null
52              && pickData.isPickChannel(PickData.P)) {
53            String channel = wvp.getChannel();
54            double distance = pickData.getSpDistance();
55            if (!Double.isNaN(distance)) {
56              spList.add(new Sp(channel, distance, pickData.getSpMinDistance(),
57                  pickData.getSpMaxDistance()));
58            }
59          }
60        }
61      }
62      return spList;
63    }
64    
65    /**
66     * @see gov.usgs.volcanoes.swarm.map.MapLayer#draw(java.awt.Graphics2D)
67     */
68    public void draw(Graphics2D g2) {
69      Color color = g2.getColor();
70      Stroke stroke = g2.getStroke();
71      Shape clip = g2.getClip();
72      g2.setColor(new Color(SwarmConfig.getInstance().mapLineColor));
73      
74      // First get updated list of S-P
75      Vector<Sp> spList = updateSpList();
76      if (spList.size() == 0) {
77        return;
78      }
79  
80      MapPanel  panel = MapFrame.getInstance().getMapPanel();
81      
82      // set clip
83      int widthPx = panel.getGraphWidth();
84      int heightPx = panel.getGraphHeight();
85      int inset = panel.getInset();
86      Rectangle2D rectangle = new Rectangle2D.Double(inset, inset, widthPx, heightPx);
87      g2.setClip(rectangle);
88      
89      // draw
90      GeoRange range = panel.getRange();
91      Projection projection = panel.getProjection();
92      final double[] ext = range.getProjectedExtents(projection);
93      final double dx = (ext[1] - ext[0]);
94      for (Sp sp : spList) {
95        Metadata md = SwarmConfig.getInstance().getMetadata(sp.channel);
96        if (md == null) {
97          continue;
98        }
99        if (md.hasLonLat()) {
100         Point2D.Double center = panel.getXy(md.getLongitude(), md.getLatitude());
101 
102         // draw S-P circle
103         g2.setStroke(solid);
104         double radius = widthPx * 1000 * sp.distance / dx;
105         g2.drawOval((int) (center.x - radius), (int) (center.y - radius), (int) radius * 2,
106             (int) radius * 2);
107         // draw S-P uncertainty
108         g2.setStroke(dashed);
109         if (!Double.isNaN(sp.minDistance)) {
110           double minRadius = widthPx * 1000 * sp.minDistance / dx;
111           g2.drawOval((int) (center.x - minRadius), (int) (center.y - minRadius),
112               (int) minRadius * 2, (int) minRadius * 2);
113         }
114         if (!Double.isNaN(sp.maxDistance)) {
115           double maxRadius = widthPx * 1000 * sp.maxDistance / dx;
116           g2.drawOval((int) (center.x - maxRadius), (int) (center.y - maxRadius),
117               (int) maxRadius * 2, (int) maxRadius * 2);
118         }
119       }
120     }
121     
122     g2.setColor(color);
123     g2.setStroke(stroke);
124     g2.setClip(clip);
125   }
