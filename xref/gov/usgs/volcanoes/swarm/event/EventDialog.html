457   /**
458    * Open file chooser dialog.
459    * 
460    * @param selectionMode file or directory
461    * @return filename
462    */
463   private String openFileChooser(int selectionMode, String filename, FileFilter filter) {
464     setAlwaysOnTop(false);
465     JFileChooser chooser = new JFileChooser();
466     chooser.setCurrentDirectory(new File(SwarmConfig.getInstance().lastPath));
467     if (filename != null) {
468       chooser.setSelectedFile(new File(filename));
469     }
470     if (filter != null) {
471       chooser.setFileFilter(filter);
472     }
473     chooser.setFileSelectionMode(selectionMode);
474     chooser.setMultiSelectionEnabled(false);
475     chooser.setDialogTitle("Open File...");
476     int result = chooser.showOpenDialog(Swarm.getApplicationFrame());
477     if (result == JFileChooser.APPROVE_OPTION) {
478       File file = chooser.getSelectedFile();
479       if (selectionMode == JFileChooser.DIRECTORIES_ONLY) {
480         SwarmConfig.getInstance().lastPath = file.getAbsolutePath();
481       } else {
482         SwarmConfig.getInstance().lastPath = file.getParent();
483       }
484       return file.getAbsolutePath();
485     }
486     setAlwaysOnTop(true);
487     setVisible(true);
488     toFront();
489     requestFocus();
490     return null;
491   }

493   /**
494    * Locate earthquake using hypo71.
495    * 
496    * @throws ParseException parse exception
497    * @throws IOException IO exception
498    */
499   private void runHypo71() throws IOException, ParseException {
500     hypoResult = null;
501     hypo71Mgr.clear();
502     boolean success;
503     if (useInputFile.isSelected()) {
504       String filename = hypo71InputFile.getText();
505       if (filename == null || filename.equals("")) {
506         JOptionPane.showMessageDialog(this, "Please select a Hypo71 input file to use.");
507         return;
508       }
509       success = hypo71Mgr.calculate(hypo71InputFile.getText());
510     } else {
511       hypo71Mgr.loadCrustalModelFromFile();
512       hypo71Mgr.description = description.getText();
513       for (WaveViewPanel wvp : WaveClipboardFrame.getInstance().getWaves()) {
514         PickData pickData = wvp.getPickData();
515         Pick p = pickData.getPick(PickData.P);
516         if (p == null || !pickData.isPickChannel(PickData.P)) {
517           continue;
518         }
519 
520         // add station
521         String channel = wvp.getChannel();
522         Metadata md = SwarmConfig.getInstance().getMetadata(channel);
523         if (md == null) {
524           String message = "No metadata found for " + channel
525               + ". It will be not be used to calculate solution.";
526           LOGGER.error(message);
527           continue;
528         }
529         String station = md.getSCNL().station;
530         Double delay = Double.isNaN(md.getDelay()) ? 0 : md.getDelay();
531         Double fmag = Double.isNaN(md.getFmagCorrection()) ? 0 : md.getFmagCorrection();
532         Double xmag = Double.isNaN(md.getXmagCorrection()) ? 0 : md.getXmagCorrection();
533         double latitude = md.getLatitude();
534         double longitude = md.getLongitude();
535         if (Double.isNaN(latitude) || Double.isNaN(longitude)) {
536           String message = "No location metadata found for " + channel
537               + ". It will be not be used to calculate solution.";
538           LOGGER.error(message);
539           continue;
540         }
541         double height = md.getHeight();
542         try {
543           hypo71Mgr.addStation(station, latitude, longitude, height, delay, fmag, xmag, 0);
544         } catch (IllegalArgumentException e) {
545           JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), e.getMessage());
546           return;
547         }
548 
549         // add phase record
550         Pick coda1 = pickData.getPick(PickData.CODA1);
551         Pick coda2 = pickData.getPick(PickData.CODA2);
552         double fmp = 0;
553         if (coda1 != null || coda2 != null) {
554           long endCoda = 0;
555           if (coda1 == null) {
556             endCoda = coda2.getTime();
557           } else if (coda2 == null) {
558             endCoda = coda1.getTime();
559           } else {
560             endCoda = Math.max(coda1.getTime(), coda2.getTime());
561           }
562           fmp = (endCoda - p.getTime()) / 1000.0;
563         }
564         Pick s = pickData.getPick(PickData.S);
565         hypo71Mgr.addPhaseRecord(station, p, s, fmp);
566       }
567       int numPhaseRecords = hypo71Mgr.phaseRecordsList.size();
568       String message = "Number of stations: " + numPhaseRecords;
569       String title = "Hypo71";
570       setAlwaysOnTop(false);
571       if (numPhaseRecords < 3) {
572         message += "\n\nA minimum of 3 stations is required for a solution.";
573         JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), message, title,
574             JOptionPane.ERROR_MESSAGE);
575         return;
576       } else {
577         JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), message, title,
578             JOptionPane.INFORMATION_MESSAGE);
579       }
580       setAlwaysOnTop(true);
581       requestFocus();
582       PhaseRecord endRecord = new PhaseRecord();
583       endRecord.setMSTA("    ");
584       hypo71Mgr.phaseRecordsList.add(endRecord);
585       setAlwaysOnTop(false);
586       success = hypo71Mgr.calculate(null);
587       setAlwaysOnTop(true);
588     }
589     if (success) {
590       hypoResult = hypo71Mgr.hypo71.getResults();
591       updateHypo71Output(hypoResult);
592       hypo71Mgr.clear();
593     }
594   }

596   /**
597    * Update hypo71 output panel with key output information.
598    * 
599    * @param results hypo71 results
600    */
601   private void updateHypo71Output(Hypo71.Results results) {
602     String output = results.getOutput();
603     hypo71Output = output;
604 
605     char ins = hypoResult.getStationsResultList().get(0).getINS();
606     char iew = hypoResult.getStationsResultList().get(0).getIEW();
607     Hypocenter hypocenter = results.getHypocenterOutput().get(0);
608     double latitude = hypocenter.getLatitude();
609     if (ins == 'S') {
610       latitude *= -1;
611     }
612     double longitude = hypocenter.getLongitude();
613     if (iew != 'E') {
614       longitude *= -1;
615     }
616     double depth = hypocenter.getZ();
617     String locationString =
618         String.format("%.2f  %.2f  %.2f km", latitude, longitude, depth);
619     location.setText(locationString);
620     magnitude.setText(hypocenter.getMAGOUT().trim());
621     rms.setText(String.format("%.2f s", hypocenter.getRMS()));
622     gap.setText(String.format("%s deg",hypocenter.getIGAP()));
623     erh.setText(hypocenter.getERHOUT().trim() + " km");
624     erz.setText(hypocenter.getSE3OUT().trim() + " km");
625   }

627   /**
628    * Create event.
629    */
630   private Event createEvent() {
631     String eventId = Long.toString(System.currentTimeMillis());
632     String publicId = QUAKEML_RESOURCE_ID + "/Event/" + eventId;
633     Event event = new Event(publicId);
634     event.setEventId(eventId);
635     event.setEventSource(QUAKEML_RESOURCE_ID);
636     event.setType((EventType) eventType.getSelectedItem());
637     event.setTypeCertainty((EventTypeCertainty) eventTypeCertainty.getSelectedItem());
638     event.setDescription(description.getText());
639     event.setComment(comment.getText());
640 
641     // add picks
642     HashMap<String, Pick> picks = new HashMap<String, Pick>();
643     if (usePicks.isSelected()) {
644       for (WaveViewPanel wvp : WaveClipboardFrame.getInstance().getWaves()) {
645         PickData pickData = wvp.getPickData();
646         for (String phase : new String[] {PickData.P, PickData.S}) {
647           if (pickData.getPick(phase) != null && pickData.isPickChannel(phase)) {
648             Pick pick = pickData.getPick(phase);
649             picks.put(pick.publicId, pick);
650           }
651         }
652         for (String phase : new String[] {PickData.CODA1, PickData.CODA2}) {
653           Pick pick = pickData.getPick(phase);
654           if (pick != null) {
655             picks.put(pick.publicId, pick);
656           }
657         }
658       }
659     }
660     event.setPicks(picks);
661 
662     // If hypo71 was run
663     if (hypoResult != null) {
664       // get origins and magnitudes from hypo71 output
665       char ins = hypoResult.getStationsResultList().get(0).getINS();
666       char iew = hypoResult.getStationsResultList().get(0).getIEW();
667       String mtype = hypo71Mgr.getMagOutType();
668       List<Hypocenter> hypocenters = hypoResult.getHypocenterOutput();
669       HashMap<String, Origin> origins = new HashMap<String, Origin>();
670       HashMap<String, Magnitude> magnitudes = new HashMap<String, Magnitude>();
671       int magCount = 0;
672       int originCount = 0;
673       for (Hypocenter hypocenter : hypocenters) {
674         // Magnitude
675         if (!hypocenter.getMAGOUT().trim().isEmpty()) {
676           double mag = Double.parseDouble(hypocenter.getMAGOUT());
677           publicId = QUAKEML_RESOURCE_ID + "/Magnitude/" + magCount;
678           Magnitude magnitude = new Magnitude(publicId, mag);
679           magnitudes.put(publicId, magnitude);
680           magnitude.setType(mtype);
681           magCount++;
682           event.setPreferredMagnitude(magnitude);
683         }
684 
685         try {
686           // Origin
687           // time
688           int kdate = hypocenter.getKDATE();
689           int khr = hypocenter.getKHR();
690           int kmin = hypocenter.getKMIN();
691           int sec = (int) hypocenter.getSEC();
692           long time;
693           time = getDate(kdate, khr, kmin, sec).getTime();
694 
695           // hypocenter
696           double latitude = hypocenter.getLatitude();
697           if (ins == 'S') {
698             latitude *= -1;
699           }
700           double longitude = hypocenter.getLongitude();
701           if (iew != 'E') {
702             longitude *= -1;
703           }
704           double depth = hypocenter.getZ() * 1000;
705 
706           publicId = QUAKEML_RESOURCE_ID + "/Origin/" + originCount;
707           Origin origin = new Origin(publicId, time, longitude, latitude);
708           origin.setDepth(depth);
709 
710           // quality
711           OriginQuality quality = new OriginQuality();
712           quality.setAzimuthalGap(hypocenter.getIGAP());
713           quality.setStandardError(hypocenter.getRMS());
714           quality.setAssociatedStationCount(hypocenter.getNR());
715           quality.setUsedPhaseCount(hypocenter.getNO());
716           double dm = hypocenter.getDMIN();
717           dm = Math.toDegrees(dm / 6371); // convert km to degrees
718           quality.setMinimumDistance(dm);
719           origin.setQuality(quality);
720 
721           origin.setEvaluationMode(EvaluationMode.MANUAL);
722           origins.put(publicId, origin);
723           event.setPreferredOrigin(origin);
724           originCount++;
725         } catch (ParseException e) {
726           // TODO Auto-generated catch block
727           e.printStackTrace();
728         }
729       }
730       if (magnitudes.size() > 0) {
731         event.setMagnitudes(magnitudes);
732       }
733       if (origins.size() > 0) {
734         event.setOrigins(origins);
735       }
736 
737       // get station results and create station magnitudes
738       HashMap<String, StationMagnitude> stationMags = new HashMap<String, StationMagnitude>();
739       HashMap<String, Station> stations = new HashMap<String, Station>();
740       int magNum = 1;
741       for (Station station : hypoResult.getStationsResultList()) {
742         stations.put(station.getNSTA(), station);
743         double fmag = station.getFMAG();
744         if (fmag > 0.0 && fmag < 99) {
745           publicId = QUAKEML_RESOURCE_ID + "/StationMagnitude/" + magNum;
746           StationMagnitude stationMag = new StationMagnitude(publicId,
747               event.getPreferredOrigin().publicId, station.getFMAG());
748           stationMag.setType("Md");
749           stationMags.put(publicId, stationMag);
750           magNum++;
751         }
752       }
753       event.setStationMagnitudes(stationMags);
754 
755       // add arrivals
756       Origin origin = event.getPreferredOrigin();
757       if (origin != null) {
758         HashMap<String, Arrival> arrivalMap = new HashMap<String, Arrival>();
759         int arrivalNum = 1;
760         for (Pick pick : picks.values()) {
761           if (!pick.getPhaseHint().equals(PickData.P) && !pick.getPhaseHint().equals(PickData.S)) {
762             continue;
763           }
764           publicId = QUAKEML_RESOURCE_ID + "/Arrival/" + arrivalNum;
765           Arrival arrival = new Arrival(publicId, pick, pick.getPhaseHint());
766           String stationName = pick.getChannel().split("\\$")[0];
767           Station station = stations.get(stationName);
768           if (station == null) {
769             continue;
770           }
771           double distanceKm = station.getDIST();
772           double distanceDeg = distanceKm / 111.32;
773           arrival.setDistance(distanceDeg);
774           arrival.setAzimuth(station.getAZI());
775           arrival.setTakeoffAngle(station.getAIN());
776           if (pick.getPhaseHint().equals(PickData.P)) {
777             arrival.setTimeResidual(station.getPRES());
778             arrival.setTimeWeight(station.getPWT());
779           }
780           if (pick.getPhaseHint().equals(PickData.S)) {
781             arrival.setTimeResidual(station.getSRES());
782             arrival.setTimeWeight(station.getSWT());
783           }
784           arrivalMap.put(publicId, arrival);
785           arrivalNum++;
786         }
787         origin.setArrivals(arrivalMap);
788       }
789     }
790 
791     return event;
792   }

795   private SimpleDateFormat df = new SimpleDateFormat("yyMMddHHmmss");
796 
797   /**
798    * Get date object from hypo71 hypocenter date format.
799    * 
800    * @param kdate year, month, and day (YYMMDD)
801    * @param khr hour
802    * @param kmin minute
803    * @param ksec seconds
804    * @return date
805    * @throws ParseException parse exception
806    */
807   private Date getDate(int kdate, int khr, int kmin, int ksec) throws ParseException {
808     String date = String.format("%06d", kdate);
809     String hour = String.format("%02d", khr);
810     String min = String.format("%02d", kmin);
811     String sec = String.format("%02d", ksec);
812     String ds = date + hour + min + sec;
813     Date d = df.parse(ds);
814     return d;
815   }

817   /**
818    * Import event from file.
819    * 
820    * @param f file
821    */
822   private void importQuakeMl(String filename) {
823     try {
824       if (filename == null) {
825         return;
826       }
827       WaveClipboardFrame clipboard = WaveClipboardFrame.getInstance();
828       EventSet eventSet = EventSet.parseQuakeml(new FileInputStream(new File(filename)));
829       if (eventSet.size() == 0) {
830         setAlwaysOnTop(false);
831         JOptionPane.showMessageDialog(clipboard, "No events found in file.");
832         setAlwaysOnTop(true);
833         requestFocus();
834         return;
835       }
836       setAlwaysOnTop(false);
837       Event event;
838       if (eventSet.size() > 1) { // Get user to decide which event to import
839         HashMap<String, Event> eventMap = new HashMap<String, Event>();
840         for (Event e : eventSet.values()) {
841           String description = e.getDescription();
842           if (description == null || description.equals("")) {
843             description = e.publicId;
844           }
845           eventMap.put(description, e);
846         }
847         event = openEventChooser(eventMap);
848       } else {
849         event = eventSet.values().iterator().next();
850       }        
851       // See if user wants to clear clipboard first
852       int result = JOptionPane.showConfirmDialog(Swarm.getApplicationFrame(),
853           "Clear clipboard first?", "Import Event", JOptionPane.YES_NO_OPTION);
854       String message = "This may take a while as it loads wave information from data sources.";
855       JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), message);
856       if (result == JOptionPane.YES_OPTION) {
857         clipboard.importEvent(event, true);
858       } else {
859         clipboard.importEvent(event, false);
860       }
861       setAlwaysOnTop(true);
862       requestFocus();
863       hypo71Output = "";
864     } catch (FileNotFoundException e) {
865       LOGGER.warn(e.getMessage());
866     } catch (IOException e) {
867       LOGGER.warn(e.getMessage());
868     } catch (ParserConfigurationException e) {
869       LOGGER.warn(e.getMessage());
870     } catch (SAXException e) {
871       LOGGER.warn(e.getMessage());
872     }
873   }

888   /**
889    * Save event to QuakeML file.
890    */
891   public void saveQuakeMl(String filename) {
892     if (filename == null) {
893       return;
894     }
895     try {
896       // build XML document
897       DocumentBuilderFactory docFactory = DocumentBuilderFactory.newInstance();
898       DocumentBuilder docBuilder = docFactory.newDocumentBuilder();
899 
900       Document doc = docBuilder.newDocument();
901       Element quakeml = doc.createElement("q:quakeml");
902       quakeml.setAttribute("xmlns:catalog", "http://xmlns/catalog/0.1");
903       doc.appendChild(quakeml);
904 
905       Element eventParameters = doc.createElement("eventParameters");
906       eventParameters.setAttribute("publicID", QUAKEML_RESOURCE_ID);
907       Event event = createEvent();
908       eventParameters.appendChild(event.toElement(doc));
909       quakeml.appendChild(eventParameters);
910 
911       Element creationInfo = doc.createElement("creationInfo");
912       Element author = doc.createElement("author");
913       author.appendChild(doc.createTextNode(user));
914       creationInfo.appendChild(author);
915       Element creationTime = doc.createElement("creationTime");
916       creationTime
917           .appendChild(doc.createTextNode(QuakeMlUtils.formatDate(System.currentTimeMillis())));
918       creationInfo.appendChild(creationTime);
919 
920       // write to file
921       FileOutputStream fos = new FileOutputStream(filename);
922 
923       TransformerFactory transformerFactory = TransformerFactory.newInstance();
924       Transformer transformer = transformerFactory.newTransformer();
925       transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
926       transformer.setOutputProperty(OutputKeys.INDENT, "yes");
927       transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "4");
928       DOMSource source = new DOMSource(doc);
929       StreamResult result = new StreamResult(fos);
930       transformer.transform(source, result);
931 
932       fos.close();
933 
934       String message = "QuakeML event saved.";
935       JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), message);
936     } catch (ParserConfigurationException e) {
937       // TODO Auto-generated catch block
938       e.printStackTrace();
939     } catch (TransformerConfigurationException e) {
940       // TODO Auto-generated catch block
941       e.printStackTrace();
942     } catch (FileNotFoundException e) {
943       // TODO Auto-generated catch block
944       e.printStackTrace();
945     } catch (TransformerException e) {
946       // TODO Auto-generated catch block
947       e.printStackTrace();
948     } catch (IOException e) {
949       // TODO Auto-generated catch block
950       e.printStackTrace();
951     }
952   }
