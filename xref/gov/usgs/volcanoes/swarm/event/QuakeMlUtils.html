23  /**
24   * A utility class for working with QuakeML files.
25   * 
26   * @author Tom Parker
27   *
28   */
29  @Deprecated
30  public class QuakeMlUtils {
31    private static final Logger LOGGER = LoggerFactory.getLogger(QuakeMlUtils.class);
32    private static final String DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSX";
33  
34    /**
35     * Format date for QuakeML
36     * 
37     * @param millis milliseconds since 1/1/1970 00:00:00 GMT
38     * @return time string in yyyy-MM-dd'T'HH:mm:ss.SSSX format
39     */
40    public static String formatDate(long millis) {
41      SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
42      Date date = new Date(millis);
43      return dateF.format(date);
44    }
45    
46    /**
47     * Parse QuakeML date/time.
48     * @param timeString time string in yyyy-MM-dd'T'HH:mm:ss.SSSX format
49     * @return milliseconds since 1/1/1970 00:00:00 GMT
50     */
51    public static long parseTime(String timeString) {
52      String inString = timeString;
53      timeString = timeString.replaceFirst("\\.(\\d)Z?", ".$100Z");
54      timeString = timeString.replaceFirst("\\.(\\d{2})Z?$", ".$10Z");
55      timeString = timeString.replaceFirst(":(\\d{2})Z?$", ":$1.000Z");
56  
57      long time = Long.MIN_VALUE;
58      SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
59      try {
60        time = dateF.parse(timeString).getTime();
61      } catch (ParseException e) {
62        LOGGER.error("Cannot parse time String {}", inString);
63        throw new RuntimeException("Cannot parse time string " + inString);
64      }
65      return time;
66    }
