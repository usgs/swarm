1   package gov.usgs.volcanoes.swarm.event;
2   
3   import java.io.FileReader;
4   import java.io.FileWriter;
5   import java.io.IOException;
6   import java.util.Properties;
7   
8   /**
9    * Pick Settings.
10   * @author Diana Norgaard
11   *
12   */
13  public class PickSettings extends Properties {
14    private static final long serialVersionUID = 897150893674347168L;
15    private static final String SETTINGS_FILENAME = "PickSettings.config";
16    public static final String WEIGHT = "weight";
17    public static final String WEIGHT_UNIT = "weight_unit";
18    public static final int numWeight = 5;
19    private static final String COMMENTS = "Swarm Pick Settings Configuration";
20    
21    public enum WeightUnit { SAMPLES, MILLISECONDS }
22    
23    
24    /**
25     * Default constructor.
26     */
27    public PickSettings() {
28      readSettings();
29    }
30  
31    /**
32     * Read pick settings file.
33     */
34    private void readSettings() {
35      loadDefaultProperties(); 
36      try {
37        load(new FileReader(SETTINGS_FILENAME));
38      } catch (IOException e) {
39        // just use default settings
40      }
41  
42    }
43  
44    /**
45     * Load default properties in cases where settings file does not exist.
46     */
47    private void loadDefaultProperties() {
48      this.setProperty(WEIGHT_UNIT, WeightUnit.SAMPLES.toString());
49      this.setProperty(WEIGHT + ".0", "1");
50      this.setProperty(WEIGHT + ".1", "2");
51      this.setProperty(WEIGHT + ".2", "5");
52      this.setProperty(WEIGHT + ".3", "10");
53      this.setProperty(WEIGHT + ".4", "20");
54    }
55    
56    /**
57     * Save pick settings properties to file.
58     * @throws IOException IOException
59     */
60    protected void save() throws IOException {
61      FileWriter writer = new FileWriter(SETTINGS_FILENAME);
62      store(writer, COMMENTS);
63    }
64     
65    /**
66     * Convert weight to time. In case where unit is in samples, 
67     * the sample rate is required.
68     * @param weight weight
69     * @param sampleRate sample rate to use if weights are in samples
70     * @return milliseconds
71     */
72    public long getWeightToTime(int weight, double sampleRate) {
73      if (weight > numWeight) {
74        String message = "Weight must be between 0 and " + numWeight;
75        throw new IllegalArgumentException(message);
76      }
77      String prop = getProperty(WEIGHT + "." + weight);
78      long value = Long.valueOf(prop);
79      if (getWeightUnit().equals(WeightUnit.SAMPLES)) {
80        long millis = (long) (1000 * value / sampleRate);
81        return millis;
82      } else {
83        return value; // already in millis
84      }
85    }
86  
87    /**
88     * Type of unit mapped to weight.
89     * @return WeightUnit
90     */
91    public WeightUnit getWeightUnit() {
92      return WeightUnit.valueOf(getProperty(WEIGHT_UNIT));
93    }
94  }
