5   public final class FileChooser {
6       private static final FileChooser INSTANCE = new FileChooser();
7       private static JFileChooser fileChooser;
8       
9       private FileChooser() {
10          Thread t = new Thread(new Runnable() {
11              public void run() {
12                  fileChooser = new JFileChooser();
13              }
14          });
15          
16          t.setPriority(Thread.MIN_PRIORITY);
17          t.start();
18      }
19      
20      public static JFileChooser getFileChooser() {
21          int timeout = 10000;
22          while (fileChooser == null && timeout > 0) {
23              try {
24                  Thread.sleep(100);
25              } catch (Exception e) {
26              }
27              timeout -= 100;
28          }
29          return fileChooser;
30      }
